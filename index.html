import json import os from datetime import datetime from typing import Dict, Any class KurdChatMemory: def __init__(self, memory_file="kurdchat_memory.json"): self.memory_file = memory_file self.memory = self._load_memory() def _load_memory(self) -> Dict: """بارکردنی هەموو یادەوەرییەکان""" if os.path.exists(self.memory_file): with open(self.memory_file, 'r', encoding='utf-8') as f: return json.load(f) return {"users": {}, "conversations": {}} def save_memory(self): """هەڵگرتنی یادەوەرییەکان""" with open(self.memory_file, 'w', encoding='utf-8') as f: json.dump(self.memory, f, ensure_ascii=False, indent=4) def register_user(self, user_id: str, name: str, preferences: Dict = None): """تۆمارکردنی بەکارهێنەری نوێ""" if user_id not in self.memory["users"]: self.memory["users"][user_id] = { "name": name, "preferences": preferences or {}, "created_at": str(datetime.now()), "conversation_history": [] } self.save_memory() return f"بەخێربێیت {name}! تۆمارکرایت لە KurdChat." return f"بەخێربێیتەوە {self.memory['users'][user_id]['name']}!" def add_conversation(self, user_id: str, user_msg: str, bot_msg: str): """زیادکردنی گفتوگۆی نوێ""" if user_id in self.memory["users"]: self.memory["users"][user_id]["conversation_history"].append({ "timestamp": str(datetime.now()), "user": user_msg, "bot": bot_msg }) self.save_memory() def get_user_info(self, user_id: str) -> Dict: """زانیاری بەکارهێنەر وەرگرتن""" return self.memory["users"].get(user_id, {}) # نمونەی بەکارهێنان memory_system = KurdChatMemory() # تۆمارکردنی بەکارهێنەر print(memory_system.register_user("user_001", "نانا")) # زیادکردنی گفتوگۆ memory_system.add_conversation("user_001", "سڵاو", "سڵاو نانا! چۆنی؟") # زانیاری بەکارهێنەر user_info = memory_system.get_user_info("user_001") print(f"بەکارهێنەر: {user_info.get('name')}") print(f"ژمارەی گفتوگۆکان: {len(user_info.get('conversation_history', []))}")
